s5 <- c("pH.peat", "P.peat", "Waterlev.av", "CEC.peat", "Acidity.peat")
ssit5 <- ssit[s5]
par(mfrow = c(1, 2))

# RDA = constrained PCA
rda <- rda(sveg ~ ., ssit5)
plot(rda)

# CCA = constrained CA
cca <- cca(sveg ~ ., ssit5)
plot(cca)


# Unconstrained and constrained variance
tot <- cca$tot.chi
constr <- cca$CCA$tot.chi
constr / tot # Erklärte Varianz
## [1] 0.2621114
